Простая и интуитивная программная модель

Кликни на подсвеченных блоках кода, чтобы увидеть иллюстрации 
{
"use strict";

$(function () {
    // RedUI сгенерирует HTML по заданной модели
    var myWindow = redui.createNewWindow("CustomerDataWindow");

    // Отобразить окно на экране
    myWindow.show();

    // Для привязки к данным требуется лишь вызвать bind
    var customerData = {
        firstName: "Claudio",
        lastName: "Lucchesi",  
    };
    myWindow.bind(customerData);

    // Элементы управления доступны по именам...
    var firstNameControl = myWindow.firstNameTextBox;

    // ...или по индексу
    var firstControl = myWindow.controls[0];

    // Получив элемент управления, можно...

    // ...получить HTML элемент
    var element = firstNameControl.getElement();
    // ...свойства модели
    var label = firstNameControl.model.label;
    // ...объект с данными, привязанный к элементу управления
    var dataObject = firstNameControl.bindingContext;

    // Элементы управления генерируют события
    myWindow.closed(function (target, dialogResult) {
        if (dialogResult === "OK") {
            // Отправить данные на сервер
        }
    });
});

Привязка к данным работает в обе стороны

Когда пользователь меняет значение на экране, привязанный объект данных обновляется автоматически.

Встроенная валидация

Поддерживаются обязательные поля, соответствие шаблону, ограничение максимальной длины, пользовательская валидация (если в привязанном объекте данных определена функция validate). См. Интерактивные примеры чтобы узнать, как задать правила валидации.

Поддержка локализации

Основано на возможностях библиотеки globalize.js (не требуется, если не нужно). См. Интерактивные примеры чтобы узнать, как локализовать элементы управления.

Supports ko.observable

Поддержка knockout.js (не требуется, если не нужно). См. Интерактивные примеры чтобы узнать, как использовать ko.observable.
